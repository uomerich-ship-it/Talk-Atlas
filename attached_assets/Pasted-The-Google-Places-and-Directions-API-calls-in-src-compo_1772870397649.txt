The Google Places and Directions API calls in 
src/components/right/WayfinderPanel.tsx are failing with 
"Failed to fetch" because Google blocks direct browser requests 
to their REST APIs due to CORS restrictions.

Fix this by creating server-side proxy routes that the frontend 
calls instead of calling Google directly.

STEP 1 — Create server-side routes.

In the server folder (wherever the existing Express routes are),
add a new file server/routes/google.ts:

import { Router } from 'express';
import fetch from 'node-fetch';

const router = Router();
const GOOGLE_KEY = process.env.VITE_GOOGLE_PLACES_KEY || 
                   process.env.GOOGLE_PLACES_KEY || '';

// Places text search proxy
router.get('/api/places/search', async (req, res) => {
  try {
    const { query } = req.query;
    if (!query) return res.status(400).json({ error: 'query required' });
    if (!GOOGLE_KEY) return res.status(501).json({ error: 'Google API key not configured' });

    const url = `https://maps.googleapis.com/maps/api/place/textsearch/json?query=${encodeURIComponent(query as string)}&key=${GOOGLE_KEY}`;
    const response = await fetch(url);
    const data = await response.json();
    res.json(data);
  } catch (err: any) {
    res.status(500).json({ error: err.message });
  }
});

// Directions proxy
router.get('/api/directions', async (req, res) => {
  try {
    const { origin, destination, mode } = req.query;
    if (!origin || !destination) return res.status(400).json({ error: 'origin and destination required' });
    if (!GOOGLE_KEY) return res.status(501).json({ error: 'Google API key not configured' });

    const url = `https://maps.googleapis.com/maps/api/directions/json?origin=${encodeURIComponent(origin as string)}&destination=${encodeURIComponent(destination as string)}&mode=${mode || 'driving'}&key=${GOOGLE_KEY}`;
    const response = await fetch(url);
    const data = await response.json();
    res.json(data);
  } catch (err: any) {
    res.status(500).json({ error: err.message });
  }
});

export default router;

STEP 2 — Register the routes in the main server file.
Find the main server file (likely server/index.ts or server/app.ts)
and add:
import googleRoutes from './routes/google';
app.use(googleRoutes);

STEP 3 — Update WayfinderPanel.tsx to call the proxy instead of Google directly.

In the handleSearch function, replace the Google fetch URL:
FROM: fetch(`https://maps.googleapis.com/maps/api/place/textsearch/json?...`)
TO:   fetch(`/api/places/search?query=${encodeURIComponent(searchQuery)}`)

Then parse the response the same way as before 
(data.results, data.status etc — same structure).

In the directions fetch, replace:
FROM: fetch(`https://maps.googleapis.com/maps/api/directions/json?...`)
TO:   fetch(`/api/directions?origin=${encodeURIComponent(origin)}&destination=${encodeURIComponent(destination)}&mode=${travelMode}`)

Then parse response.routes[0].legs[0] exactly as before.

STEP 4 — Also add GOOGLE_PLACES_KEY (without VITE_ prefix) 
to the server environment check since server-side code cannot 
read VITE_ prefixed variables. The server route already checks 
both VITE_GOOGLE_PLACES_KEY and GOOGLE_PLACES_KEY so just make 
sure both are in Replit Secrets.
